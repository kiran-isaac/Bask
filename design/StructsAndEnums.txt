struct Vector (
  float x, float y
).{
  fn length() -> float {
    Math.squrt(this.x ** 2, this.y ** 2)
  }

  fn arg() -> float {
    Math.Arctan(this.y / this.x)
  }
}

enum Vectors with Vector (
  I(new Vector(1., 0.)),
  J(new Vector(0., 1.))
).{
  fn length() -> float {
    1.
  }

  static fn select(int i) -> This.0 { // This.0 is compile time access to the first type parameter
    if i == 0 {
      Vectors.I
    } else {
      Vectors.J
    }
  } 
}
